int kthSmallest(vector<int>& arr, int k)
{
    // Sort the given vector
    sort(arr.begin(), arr.end());

    // Return k'th element in the sorted vector
    return arr[k - 1];
}

//Driver Code Starts
int main()
{
    vector<int> arr = {10, 5, 4, 3, 48, 6, 2, 33, 53, 10};
    int k = 4;

    cout << kthSmallest(arr, k);
    return 0;
}
