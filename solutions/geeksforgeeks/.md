class Solution {
  public:
    int maxSubarraySum(vector<int>& arr, int k) {
        // code here
    if(arr.size()<k)
    return 0;
    int n=arr.size();
    int low=0,high=k-1;
    int sum=0;
    for(int i=low;i<=high;i++) //1st window sum 
    sum+=arr[i];
    int ans=INT_MIN;
    while(high<n){
        ans=max(ans,sum);
        low++;
        high++;
        if(high==n)
        break;
        sum=sum-arr[low-1]+arr[high];
    }
    return ans;
    }
};
