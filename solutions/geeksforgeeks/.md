class Solution {
  public:
  bool helper(vector<int>arr,int k,int mid){
      int sum=0,c=1;
      for(int i=0;i<arr.size();i++){
        if(sum+arr[i]<mid)
        sum+=arr[i];
        else{
            c++;
            sum=arr[i];
            if(c>k)
            return false;
        }
          
      }
        return true;
  }
    int findPages(vector<int> &arr, int k) {
        // code here
        int mini=INT_MIN,sum=0;
        for(int i=0;i<arr.size();i++){
            sum+=arr[i];
            mini=max(mini,arr[i]);
        }
        if(k>arr.size())
        return -1;
        int low=mini,high=sum,res=-1;
        while(low<=high){
            int mid=low+(high-low)/2;
            if(helper(arr,k,mid))
            {
                res=mid;
                high=mid-1;
            }
            else
            low=mid+1;
        }return res;
    }
};
