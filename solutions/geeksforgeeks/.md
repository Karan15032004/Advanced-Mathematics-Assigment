## Solution

class Solution {
  public:
    int countTriplets(int sum, vector<int>& arr) {
        // code here
        int ans=0;
        sort(arr.begin(),arr.end());
        for(int i=0;i<arr.size();i++){
            if(i>0&&arr[i]==arr[i-1])
            continue;
            int l=i+1,r=arr.size()-1;
            if(arr[i]+arr[l]+arr[r]<sum){
                ans+=(r-l);
                l++;
                while(l<arr.size()&&arr[l]==arr[l-1])
                l++;
            }
            else{
                r--;
            }
        }
        return ans;
    }
};
