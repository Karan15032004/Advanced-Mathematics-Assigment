class Solution {
  public:
    int countFreq(vector<int>& nums, int target) {
        // code here
        int low=0,n=nums.size()-1,high=n-1,first=-1,last=-1;
        if(n==0)
        return 0;
        while(low<=high){
            int mid=(low+high)/2;
            if(nums[mid]>target)
            high=mid-1;
            else if(nums[mid]<target)
            low=mid+1;
            else{
                first=mid;
                high=mid-1;
            }
        }
        low=0,high=n-1;
        while(low<=high){
            int mid=(low+high)/2;
            if(nums[mid]>target)
            high=mid-1;
            else if(nums[mid]<target)
            low=mid+1;
            else{
                last=mid;
                low=mid+1;
            }
        }
        if(low==-1&&high==-1)
        return 0;
        else
        return (last-first+1);
    }
};
