## Solution

class Solution {
  public:
    int findCeil(vector<int>& arr, int x) {
        // code here
        int low=0,high=arr.size()-1;
        int n=arr.size();
        int res=-1;
        while(low<=high){
            int mid=(low+high)/2;
            if(arr[mid]==x)
            return mid;
            else if(arr[mid]>x){
                res=mid;
                high=mid-1;
            }
            else
            low=mid+1;
        }
        return res;
    }
};
